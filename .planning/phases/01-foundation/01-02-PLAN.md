```typescript
import i18next from 'i18next';
import { initReactI18next } from 'react-i18next';

// Mock i18next for tests - use simple English strings

    lng: 'en',

    ns: 'common',

    missingKeyHandler: () => '',
  });
});
```

**In src/testing/test-utils.tsx, update the render wrapper:**

```typescript
import { I18nextProvider } from 'react-i18next';
import i18next from 'i18next';

// Add i18n provider to custom render
const customRender = (
  ui: React.ReactElement,
  options?: Omit<RenderOptions, 'wrapper'>,
) =>
  render(ui, {
    wrapper: ({ children }) => (
      <I18nextProvider i18n={i18next}>
        {/* Add other providers as needed (QueryClient, Router, etc.) */}

      </I18nextProvider>
    ),
    ...options,
  });

export * from '@testing-library/react';
export { customRender as render };
```

- ✅ Tests don't load JSON files (faster, no I/O overhead)
- ✅ Components can use useTranslation() hook without errors
- ✅ Missing translation keys return empty string (safe for unit testing)
- ✅ Tests focus on component logic, not translation content
- ✅ Phase 5 E2E tests will verify actual translations
